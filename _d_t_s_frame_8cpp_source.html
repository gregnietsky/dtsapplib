471 int DTSFrame::pwevent(struct dtsgui *dtsgui, dtsgui_pane p, int type, int event, void *data) {
472  struct basic_auth *auth;
473  DTSDialog *dp = (DTSDialog*)p;
474 
475  if (event == wx_PANEL_EVENT_BUTTON_YES) {
476  if (!(auth = (struct basic_auth*)dp->GetUserData())) {
477  return 0;
478  }
479  if (auth->user) {
480  free((void*)auth->user);
481  auth->user = NULL;
482  }
483  if (auth->passwd) {
484  memset((void*)auth->passwd, 0, strlen(auth->passwd));
485  free((void*)auth->passwd);
486  auth->passwd = NULL;
487  }
488 
489  auth->user = dp->FindValue("uname");
490  auth->passwd = dp->FindValue("pwd");
491  objunref(auth);
492  }
493  return 1;
494 }
