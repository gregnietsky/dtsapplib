251 void network_config_dns_domain_server_new(dtsgui_pane p, dtsgui_treeview self, dtsgui_treenode node, void *data) {
252  const char *domain;
253  char xpath[PATH_MAX];
254  int len, flg;
255 
256  if (!(domain = dtsgui_treenodeparent(node))) {
257  return;
258  }
259 
260  snprintf(xpath, PATH_MAX, "/config/DNS/Hosted/Domain[@domain = '%s']", domain);
261  free((void*)domain);
262 
263  len = strlen(xpath)+1;
264  domain = objalloc(len, NULL);
265  memcpy((void*)domain, xpath, len);
266 
267  dtsgui_textbox(p, "Name Server IP Address", "master", "", NULL);
268 
269  flg = DTS_TREE_NEW_NODE_DELETE | DTS_TREE_NEW_NODE_SORT;
270  dtsgui_newxmltreenode(self, p, node, xpath, "NameServer", "master", "master", DTS_NODE_NETWORK_CONFIG_DNS_DOMAIN_SERVER,
271  flg, NULL, (void*)domain, network_config_dns_domain_server);
