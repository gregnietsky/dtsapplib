41 static void set_temp_xml(struct xml_doc *xmldoc) {
42  struct xml_search *xs;
43  struct xml_node *xn, *xn2;
44  const char *hdir, *sdir, *extif, *conn;
45  int domc;
46 
47  xml_createpath(xmldoc, "/config/tmp/dcon");
48  xml_createpath(xmldoc, "/config/tmp/extif");
49 
50  xs = xml_xpath(xmldoc, "/config/FileServer", NULL);
51  xn = xml_getfirstnode(xs, NULL);
52  objunref(xs);
53 
54  hdir = xml_getattr(xn, "homedir");
55  sdir = xml_getattr(xn, "sharedir");
56 
57  if (hdir && sdir && strcmp(hdir, "") && strcmp(sdir, "")) {
58  domc = 1;
59  } else {
60  domc = 0;
61  }
62 
63  objunref(xn);
64 
65  xs = xml_xpath(xmldoc, "/config/tmp/dcon", NULL);
66  xn = xml_getfirstnode(xs, NULL);
67  objunref(xs);
68 
69  if (domc) {
70  xml_modify(xmldoc, xn, "1");
71  } else {
72  xml_modify(xmldoc, xn, "0");
73  }
74 
75  objunref(xn);
76 
77  xs = xml_xpath(xmldoc, "/config/tmp/extif", NULL);
78  xn2 = xml_getfirstnode(xs, NULL);
79  objunref(xs);
80 
81  xs = xml_xpath(xmldoc, "/config/IP/SysConf/Option[@option = 'External']", NULL);
82  xn = xml_getfirstnode(xs, NULL);
83  objunref(xs);
84 
85  extif = strdup(xn->value);
86  xml_setattr(xmldoc, xn2, "External", extif);
87  objunref(xn);
88 
89  xs = xml_xpath(xmldoc, "/config/IP/Dialup/Option[@option = 'Connection']", NULL);
90  xn = xml_getfirstnode(xs, NULL);
91  objunref(xs);
92 
93  conn = strdup(xn->value);
94  xml_setattr(xmldoc, xn2, "Connection", conn);
95  if (!strcmp(xn->value, "ADSL")) {
96  xml_modify(xmldoc, xn2, extif);
97  xml_setattr(xmldoc, xn2, "pppoe", "ADSL");
98  } else {
99  if (!strcmp(extif, "Dialup")) {
100  xml_modify(xmldoc, xn2, conn);
101  } else {
102  xml_modify(xmldoc, xn2, extif);
103  }
104  xml_setattr(xmldoc, xn2, "pppoe", "Dialup");
105  }
106  objunref(xn);
107  objunref(xn2);
108 
109  if (extif) {
110  free((void*)extif);
111  }
112  if (conn) {
113  free((void*)conn);
114  }
115 }

122 static void rem_temp_xml(struct xml_doc *xmldoc) {
123  struct xml_search *xs;
124  struct xml_node *xn, *xn2, *xn3;
125 
126  xs = xml_xpath(xmldoc, "/config/tmp/dcon", NULL);
127  xn = xml_getfirstnode(xs, NULL);
128  objunref(xs);
129 
130  xs = xml_xpath(xmldoc, "/config/FileServer", NULL);
131  xn2 = xml_getfirstnode(xs, NULL);
132  objunref(xs);
133 
134  if (xn && xn->value && !strcmp("1", xn->value)) {
135  xml_setattr(xmldoc, xn2, "homedir", "U");
136  xml_setattr(xmldoc, xn2, "sharedir", "S");

139  xml_setattr(xmldoc, xn2, "sharedir", "");
140  }
141 
142  objunref(xn);
143  objunref(xn2);
144 
145  xs = xml_xpath(xmldoc, "/config/tmp/extif", NULL);
146  xn = xml_getfirstnode(xs, NULL);
147  objunref(xs);
148 
149  xs = xml_xpath(xmldoc, "/config/IP/SysConf/Option[@option = 'External']", NULL);
150  xn2 = xml_getfirstnode(xs, NULL);
151  objunref(xs);
152 
153  xs = xml_xpath(xmldoc, "/config/IP/Dialup/Option[@option = 'Connection']", NULL);
154  xn3 = xml_getfirstnode(xs, NULL);
155  objunref(xs);
156 
157  if (!strcmp(xn->value, "3G") || !strcmp(xn->value, "3GIPW") || !strcmp(xn->value, "Dialup") || !strcmp(xn->value, "Leased")) {
158  xml_modify(xmldoc, xn2, "Dialup");
159  xml_modify(xmldoc, xn3, xn->value);
160  } else {
161  xml_modify(xmldoc, xn2, xn->value);
162  xml_modify(xmldoc, xn3, xml_getattr(xn, "pppoe"));
163  }
164  objunref(xn);
165  objunref(xn2);
166  objunref(xn3);
167 
168  xs = xml_xpath(xmldoc, "/config/tmp", NULL);
169  xn = xml_getfirstnode(xs, NULL);
170  objunref(xs);
171  xml_delete(xn);
172  objunref(xn);
173 }
